####
-- Clone existing directory and push changes
####
1. git clone {url}
2. git status
3. git add {file name}, {.}, {-A}
4. git commit -m "{message}"
5. git push
6. git pull

####
-- Push existing project to directory
####
1. git init
2. git add .
3. git commit -m "{message}"
4. git remote add origin {remote repository URL}
5. git push -u origin master
